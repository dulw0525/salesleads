#### 2.3.2 支付处理流程

**后端处理逻辑**:
```python
def purchase_lead(user_id, lead_id):
    # 1. 校验用户状态
    if not user.is_authenticated:
        return error("请先登录")
    if not user.is_verified:
        return error("请完成实名认证")

    # 2. 获取线索信息
    lead = Lead.get_by_id(lead_id)
    if lead.status != 'on_sale':
        return error("该线索已售出或下架")

    # 3. 校验余额
    account = Account.get_by_user(user_id)
    if account.recharge_balance < lead.price:
        return error("余额不足")

    # 4. 开始事务
    with transaction.atomic():
        # 5. 扣减买家充值分
        account.recharge_balance -= lead.price
        account.save()

        # 6. 增加卖家收益分(待结算)
        seller_account = Account.get_by_user(lead.seller_id)
        seller_account.frozen_balance += lead.price
        seller_account.save()

        # 7. 创建订单
        order = Order.create(
            order_id=generate_order_id(),
            buyer_id=user_id,
            lead_id=lead_id,
            amount=lead.price,
            status='paid'
        )

        # 8. 更新线索状态
        lead.status = 'sold'
        lead.buyer_id = user_id
        lead.sold_at = now()
        lead.save()

        # 9. 计算售后截止时间
        order.after_sale_deadline = now() + timedelta(hours=24)
        order.save()

        # 10. 记录日志
        log_transaction(user_id, 'purchase', lead.price, lead_id)

    # 11. 发送通知
    notify_seller(lead.seller_id, '您的线索已售出')
    notify_buyer(user_id, '购买成功,联系方式已解锁')

    return success(order_id)
```

#### 2.4.2 隐私号外呼流程

**调用隐私号接口**:
```python
def make_private_call(buyer_id, lead_id):
    # 1. 校验订单状态
    order = Order.get_by_lead(lead_id)
    if order.status != 'paid':
        return error("订单状态异常")

    # 2. 获取客户真实号码
    lead = Lead.get_by_id(lead_id)
    real_number = decrypt(lead.customer_phone)

    # 3. 调用隐私号API分配虚拟号
    virtual_number = privacy_call_api.allocate_number(
        caller=buyer_id,
        callee=real_number,
        record=True  # 启用录音
    )

    # 4. 发起呼叫
    call_session = privacy_call_api.make_call(
        virtual_number=virtual_number,
        duration_limit=3600  # 1小时通话限制
    )

    # 5. 创建通话记录
    CallRecord.create(
        order_id=order.id,
        buyer_id=buyer_id,
        virtual_number=virtual_number,
        start_time=now(),
        status='calling'
    )

    return success({
        'virtual_number': virtual_number,
        'call_session_id': call_session.id,
        'record_enabled': True
    })
```

def evaluate_complaint(complaint_id):
    complaint = Complaint.get_by_id(complaint_id)
    evidence_score = analyze_evidence(complaint)  # 证据评分 0-100
    seller_credit = User.get_by_id(complaint.seller_id).credit_score
    buyer_credit = User.get_by_id(complaint.buyer_id).credit_score

    # 综合评分
    total_score = (
        evidence_score * 0.6 +      # 证据权重60%
        (100 - seller_credit) * 0.2 +  # 卖家信用权重20%
        (buyer_credit - 50) * 0.2    # 买家信用权重20%
    )

    # 判定结果
    if total_score >= 70:
        return "buyer_wins"  # 买家胜诉
    elif total_score <= 30:
        return "seller_wins"  # 卖家胜诉
    else:
        return "partial_responsibility"  # 部分责任
```

def execute_partial_responsibility(complaint_id, refund_percentage):
    """
    refund_percentage: 退款比例,如50表示50%
    """
    complaint = Complaint.get_by_id(complaint_id)
    order = Order.get_by_id(complaint.order_id)
    lead = Lead.get_by_id(order.lead_id)

    refund_amount = order.amount * refund_percentage / 100
    seller_retain_amount = order.amount - refund_amount

    with transaction.atomic():
        # 1. 部分退款给买家
        buyer_account = Account.get_by_user(order.buyer_id)
        buyer_account.recharge_balance += refund_amount
        buyer_account.save()

        # 2. 卖家获得剩余金额
        seller_account = Account.get_by_user(lead.seller_id)
        seller_account.frozen_balance -= order.amount
        seller_account.income_balance += seller_retain_amount
        seller_account.save()

        # 3. 双方扣减少量信用分
        buyer = User.get_by_id(order.buyer_id)
        buyer.credit_score -= 3
        buyer.save()

        seller = User.get_by_id(lead.seller_id)
        seller.credit_score -= 3
        seller.save()

        # 4. 更新订单状态
        order.refund_status = 'partial_refunded'
        order.refund_amount = refund_amount
        order.save()

        # 5. 更新申诉状态
        complaint.status = 'partial_approved'
        complaint.save()

    # 6. 发送通知
    notify_user(order.buyer_id, f'申诉部分成立,退款{refund_amount}分')
    notify_user(lead.seller_id, f'申诉部分成立,获得{seller_retain_amount}分')
    notify_admin(f'申诉仲裁完成 - 部分责任', complaint.id)
```

## 四、异常处理流程

### 4.1 支付失败

**失败原因**:
- 余额不足
- 网络超时
- 第三方支付接口异常

**处理流程**:
```
支付失败
    ↓
判断失败原因
    ↓
余额不足 → 提示充值,跳转充值页
网络超时 → 提示重试,保留订单
接口异常 → 提示稍后重试,记录日志
```

### 4.2 空号检测失败

**失败原因**:
- 第三方接口不可用
- 网络超时
- 接口返回异常

**处理流程**:
```
空号检测失败
    ↓
记录失败日志
    ↓
标记为"待核实"
    ↓
允许继续上架
    ↓
后台异步重试
    ↓
核实失败 → 标记线索为高风险
```

### 4.3 申诉超时

**超时场景**:
- 卖家48小时未反证
- 客服3个工作日未处理

**处理规则**:
```python
# 卖家反证超时
if seller_response_deadline < now() and not seller_responded:
    complaint.status = 'seller_timeout'
    proceed_to_arbitration()

# 客服处理超时
if arbitration_deadline < now() and complaint.status == 'pending_arbitration':
    escalate_to_supervisor()
    send_escalation_notification()
```
